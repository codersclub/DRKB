Переместиться в конец файла
===========================

::: {.date}
01.01.2007
:::

    { прыгаем в конец (eof) }
    procedure gotoeof (f : file);
    begin
      { перемещаемся в начало }
      seek (f, 0);
      { перемещаемся вперед на "x" количество байт,
        в нашем случае это размер файла! }
      seek (f, filesize(f));
    end; {gotoeof}
