    function RandomGamma(k, a: Extended): Extended;
    { гамма распределение }
    var
      tr: Extended;
      i: Integer;
    begin
      tr := 1;
      for i := 1 to Round(k) do tr := tr * random;
      Result := -ln(tr) / a
    end; 

Пример использования:

    x := RandomGamma(10, 5);
