    function GetNumScrollLines: Integer;
     begin
       SystemParametersInfo(SPI_GETWHEELSCROLLLINES, 0, @Result, 0);
     end;
     
     procedure TForm1.Button1Click(Sender: TObject);
     begin
       ShowMessage(IntToStr(GetNumScrollLines));
     end;
