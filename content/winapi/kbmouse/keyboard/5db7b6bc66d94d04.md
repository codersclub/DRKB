    procedure TForm1.WMKeyDown(var Message: TWMKeyDown);
    begin
      inherited;
      case Message.CharCode of
        VK_RETURN:
          begin // ENTER pressed
            if (Message.KeyData and $1000000 <> 0) then
              begin
                      { ENTER on numeric keypad }
              end
            else
              begin
                      { ENTER on the standard keyboard }
              end;
          end;
      end;
    end;
