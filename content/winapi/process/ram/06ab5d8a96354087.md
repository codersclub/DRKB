    function GetMemoryTotalPhys : DWord; 
    var 
    memStatus: TMemoryStatus; 
    begin 
    memStatus.dwLength := sizeOf ( memStatus ); 
    GlobalMemoryStatus ( memStatus ); 
    Result := memStatus.dwTotalPhys; 
    end;
