    function StringToPWide( sStr: string; var iNewSize: integer ): PWideChar;
    var
      pw : PWideChar;
      iSize : integer;
    begin
      iSize := Length( sStr ) + 1;
      iNewSize := iSize * 2;
     
      pw := AllocMem( iNewSize );
     
      MultiByteToWideChar( CP_ACP, 0, PChar(sStr), iSize, pw, iNewSize );
     
      Result := pw;
    end;
