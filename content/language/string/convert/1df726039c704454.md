      function Pack(ch1, ch2: Char): Char;
      begin
        Assert((ch1 >= '0') and (ch1 <= '9'));
        Assert((ch2 >= '0') and (ch2 <= '9'));
          {Ord('0') is $30, so we can just use the low nybble of the character
           as value.}
        Result := Chr((Ord(ch1) and $F) or ((Ord(ch2) and $F) shl 4))
      end;
