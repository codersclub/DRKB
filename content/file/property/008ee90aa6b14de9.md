    function GetFileDate(FileName: string): string;
      var FHandle: Integer;
    begin
      FHandle := FileOpen(FileName, 0);
    try
      Result := DateTimeToStr(FileDateToDateTime(FileGetDate(FHandle)));
    finally
      FileClose(FHandle);
    end;
    end;
