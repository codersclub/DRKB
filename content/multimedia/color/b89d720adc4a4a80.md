Как определить цвет произвольной точки экрана?
==============================================

::: {.date}
01.01.2007
:::

    var
    DC: HDC;
    Color: Cardinal;
    begin
    DC := CreateDC ('MONITOR', nil, nil, nil);
    Color := GetPixel(DC, 300, 300);
    DeleteDC(DC);
    end;
