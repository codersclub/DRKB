      var
      sKey: string;

      Reg: TRegistry;

      Reg := TRegistry.Create;
      try
        Reg.RootKey := HKEY_LOCAL_MACHINE;
        sKey := format(cKEY, [GuidToString(ClassID)]);
        if Reg.OpenKey(sKey, False) then
        begin
          sComServer := Reg.ReadString('');
          if FileExists(sComServer) then

          end;
        end;
      finally
        Reg.free;
